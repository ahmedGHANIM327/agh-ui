# Usage Guide

This guide covers how to install, configure and use AGH UI components in a React application.

---

## Installation

```bash
npm install @aghanim1206/ui-react
# or
pnpm add @aghanim1206/ui-react
```

---

## Setup

### 1. Import the stylesheet

Import the CSS once in your app entry point. It contains all design tokens (CSS variables) and theme definitions.

```ts
// main.tsx / index.tsx
import '@aghanim1206/ui-react/style.css';
```

### 2. Wrap with providers

```tsx
import { ThemeProvider, ToastProvider } from '@aghanim1206/ui-react';

createRoot(document.getElementById('root')!).render(
  <ThemeProvider defaultTheme="light" global>
    <ToastProvider>
      <App />
    </ToastProvider>
  </ThemeProvider>
);
```

`ThemeProvider` — manages the active theme (applies CSS classes on `<html>` in `global` mode).

`ToastProvider` — required only if you use the `toast()` function.

## Component API rules

### Variants

Variants describe the visual intent of a component.

```tsx
<Button variant="default" />
<Button variant="primary" />
<Button variant="outline" />
<Button variant="destructive" />
<Button variant="ghost" />
<Button variant="link" />

<Badge variant="default" />
<Badge variant="primary" />
<Badge variant="outline" />
<Badge variant="destructive" />
```

### Sizes

```tsx
<Button size="sm" />
<Button size="default" />
<Button size="lg" />
<Button size="icon" iconName="settings" />
```

### States

```tsx
<Button disabled />
<Button isLoading label="Saving…" />

<Input error="This field is required." />
<Input disabled />

<Checkbox disabled />
<Checkbox error="You must accept the terms." />
```

### Icons

All components that accept icons use the `iconName` prop (typed as `IconName` — IDE autocomplete):

```tsx
<Button iconName="plus" label="Add item" />
<Button iconName="trash" variant="destructive" label="Delete" />
<Badge  iconName="check" label="Verified" />
<Icon   name="alertTriangle" size={20} />
```

## Forms

### Input

```tsx
<Input
  label="Email"
  type="email"
  placeholder="you@example.com"
  description="We'll never share your email."
/>

<Input
  label="Password"
  type="password"
/>

<Input
  label="Username"

```

> `Input` handles password visibility toggle automatically when `type="password"`.

### Textarea

```tsx

```

### Select

```tsx
<Select
```

### Checkbox & CheckboxGroup

```tsx

  label="Notifications"
  description="Choose how you want to be notified."

    { value: 'email', label: 'Email' },

  ]}
  value={selected}

<CheckboxGroup
  label="Permissions"
  options={permissionOptions}

  value={selected}
```

### Radio & RadioGroup

```tsx
<RadioGroup
  name="plan"
  label="Subscription plan"
  description="You can change this later."
  options={[
    { value: 'free',  label: 'Free',  description: 'Up to 3 projects.' },
    { value: 'pro',   label: 'Pro',   description: 'Unlimited projects.' },
    { value: 'team',  label: 'Team',  description: 'Collaborative workspace.' },
  ]}
  value={plan}
  onChange={setPlan}
/>
```

### Switch

```tsx
// Default — no border, no background (floating toggle)
<Switch label="Enable notifications" description="Receive push notifications." />

// Outline — bordered card
<Switch variant="outline" label="Dark mode" description="Switch to a dark theme." />

// Primary — primary-tinted card (great for active/featured settings)
<Switch variant="primary" label="Airplane mode" defaultChecked />

// Error state
<Switch variant="outline" label="Accept terms" error="You must accept the terms to continue." />

// Disabled
<Switch variant="outline" label="Locked setting" description="Managed by your organization." disabled />

// Bare (no label / no description — inline toggle only)
<Switch />
<Switch defaultChecked />
```

| Prop | Type | Default |
|---|---|---|
| `variant` | `"default"` \| `"outline"` \| `"primary"` | `"default"` |
| `label` | `string` | — |
| `description` | `string` | — |
| `error` | `string` | — |
| `disabled` | `boolean` | `false` |
| `checked` | `boolean` | — |
| `defaultChecked` | `boolean` | — |
| `containerClassName` | `string` | — |
| `labelClassName` | `string` | — |

> Extends all native `<input type="checkbox">` HTML attributes. When neither `label` nor `description` is provided, renders as a bare inline toggle.

## Theming at runtime

### ThemeProvider + useTheme

```tsx
import { ThemeProvider, useTheme } from '@aghanim1206/ui-react';
import type { ThemeName } from '@aghanim1206/ui-react';

// Wrap once at the root
<ThemeProvider defaultTheme="light" global>
  <App />
</ThemeProvider>

// Switch from any child component
function ThemeSwitcher() {
  const { theme, setTheme } = useTheme();

  const themes: { name: ThemeName; label: string }[] = [
    { name: 'light',      label: '☀️ Light'       },
    { name: 'dark',       label: '🌙 Dark'         },
    { name: 'green',      label: '🌿 Green'        },
    { name: 'greenDark',  label: '🌲 Green Dark'   },
    { name: 'purple',     label: '💜 Purple'       },
    { name: 'purpleDark', label: '🌌 Purple Dark'  },
  ];

  return (
    <div style={{ display: 'flex', gap: '8px', flexWrap: 'wrap' }}>
      {themes.map(({ name, label }) => (
        <Button
          key={name}
          label={label}
          variant={theme === name ? 'primary' : 'outline'}
          size="sm"
          onClick={() => setTheme(name)}
        />
      ))}
    </div>
  );
}
```

## Accessibility guidelines

Always provide a `label` for form fields — never rely on `placeholder` alone:

```tsx
// ✅ Correct
<Input label="Email" placeholder="you@example.com" />

// ❌ Avoid
<Input placeholder="Email" />
```

Always provide an error message when a field is invalid:

```tsx
<Input  label="Email"   error="A valid email is required." />
<Select label="Country" error="Please select a country." />
<CheckboxGroup label="Options" options={opts} error="Select at least one." />
```

Use `aria-label` on icon-only buttons:

```tsx
<Button size="icon" iconName="x" aria-label="Close dialog" />
```

Wrap disabled buttons in a `<span>` when inside a `Tooltip` (disabled elements don't fire mouse events):

```tsx
<Tooltip content="No permission to publish">
  <span>
    <Button disabled label="Publish" />
  </span>
</Tooltip>
```

## Complete example

```tsx
import {
  Button, Card, Input, Select, Textarea,
  CheckboxGroup, Typography, Badge,
  ToastProvider, toast, ThemeProvider,
} from '@aghanim1206/ui-react';
import '@aghanim1206/ui-react/style.css';

function ContactForm() {
  const [notifs, setNotifs] = useState<string[]>([]);

  return (
    <Card>
      <Typography variant="h3">Contact us</Typography>
      <Typography variant="p-sm" color="muted">
        We'll get back to you within 24 hours.
      </Typography>

      <Input label="Full name"    placeholder="Jane Doe" />
      <Input label="Email"        type="email" placeholder="jane@example.com" />

      <Select
        label="Subject"
        placeholder="Choose a subject"
        options={[
          { label: 'General inquiry', value: 'general' },
          { label: 'Technical issue', value: 'tech'    },
          { label: 'Billing',         value: 'billing' },
        ]}
      />

      <Textarea label="Message" placeholder="Describe your issue…" rows={4} />

      <CheckboxGroup
        label="Notify me by"
        options={[
          { value: 'email', label: 'Email' },
          { value: 'sms',   label: 'SMS'   },
        ]}
        value={notifs}
        onChange={setNotifs}
      />

      <div style={{ display: 'flex', justifyContent: 'flex-end', gap: '8px' }}>
        <Button variant="outline" label="Cancel" />
        <Button
          variant="primary"
          label="Send message"
          iconName="send"
          onClick={() => toast('Message sent!', { type: 'success' })}
        />
      </div>
    </Card>
  );
}

export default function App() {
  return (
    <ThemeProvider defaultTheme="light" global>
      <ToastProvider>
        <ContactForm />
      </ToastProvider>
    </ThemeProvider>
  );
}
```
